311 /*----------------------------------------------------------------------------*
312  * PURPOSE
313  * Returns the results of a service search attribute request. One primitive
314  * is issued for each service record received from the server, containing
315  * the attribute list in 'attr_list' and its size in 'size_attr_list'.
316  * The total size of the entire response (over all services and primitives)
317  * is contained in 'total_response_size'.
318  *
319  * NOTE - The initial Data Element Size descriptor, which contains the
320  * overall size of the service record, is stripped off by SDP and is
321  * not included in the 'attr_list' data.
322  *
323  * SDP collates the responses arriving from the server and sends only
324  * complete service records to the application. If a service record turns
325  * out to be too large for the available memory, the issued primitive will
326  * have 'size_attr_list = 0' and 'attr_list = NULL', and 'response' will
327  * contain the error SDC_RESPONSE_OUT_OF_MEMORY. SDP will discard the
328  * large service record, and will continue to handle subsequent records.
329  * Therefore the application will continue to receive primitives for the
330  * other service records, memory permitting.
331  *
332  * The 'more_to_come' flag indicates to the application whether there are
333  * more service records to come. When the last service record is sent,
334  * 'more_to_come' is set to FALSE.
335  *
336  * Note that SDP may stop issuing service records if it detects an error,
337  * this is indicated to the application by setting 'more_to_come = FALSE'
338  * and setting the 'response' and 'err_code' fields accordingly. The
339  * 'response' field indicates locally-detected error conditions, such as
340  * invalid responses from the server; the 'err_code' field indicates errors
341  * reported from the server.
342  *
343  *---------------------------------------------------------------------------*/
344 typedef struct
345 {
346  sdc_prim_t type; /* Always */
347  /* SDC_SERVICE_SEARCH_ATTRIBUTE_CFM */
348  phandle_t phandle; /* as supplied in REQ */
349  uint16_t con_ctx; /* Reserved - shall be set to 0 */
350  BD_ADDR_T bd_addr;
351 
352  /* The total size of the entire response in bytes. Note that SDP will
353  * strip out the initial DES descriptor of each service record, so
354  * applications should not expect to receive this many bytes. This
355  * value is purely informational.
356  */
357  uint16_t total_response_size;
358 
359  /* The attribute list and size for a single service, described as a data
360  * element sequence. For multiple services, this primitive will be
361  * issued multiple times, each one containing the attribute data for
362  * one service.
363  */
364  uint16_t size_attr_list;
365  uint8_t *attr_list;
366 
367  /* Non-zero if there are still some more service records to come. FALSE
368  * means that the query has finished.
369  */
370  bool_t more_to_come;
371 
372  uint16_t response; /* Indicates whether or not the */
373  /* search was successful */
374  uint16_t err_code; /* Error code if one has occurred*/
375  uint16_t size_err_info; /* size of the error info */
376  uint8_t *err_info; /* Error-specific information */
