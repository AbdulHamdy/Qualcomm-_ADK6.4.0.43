691 /*----------------------------------------------------------------------------*
692  * API:
693  * ATT_CONNECT_REQ_T
694  *
695  * PURPOSE:
696  * Primarily as a request to establish an ATT bearer between a client & a server
697  *
698  * -----------------------------------------------------------------------------
699  * "LE Advertising module related behavior":
700  * -----------------------------------------------------------------------------
701  * Bits-2:4 are special connection flags and imply the following:
702  * < Binary > < Description >
703  * xxxxxxxx xxx000xx Channel to be used for UCD
704  * xxxxxxxx xxx001xx Channel to be used for broadcast connectionless
705  * xxxxxxxx xxx010xx Connect as Master
706  * xxxxxxxx xxx011xx Connect as Master using whitelist
707  * xxxxxxxx xxx100xx Start high duty cycle directed adverts
708  * xxxxxxxx xxx101xx Use whitelist for undirected adverts
709  * xxxxxxxx xxx110xx Start undirected adverts
710  * xxxxxxxx xxx111xx Start low duty cycle directed adverts
711  *
712  * If its used to enable LE connectable advertising (using above connection
713  * flags) then it won't fail because of invalid advertising parameter values.
714  * Hence application's request to enable LE advertising using this API would
715  * always succeed provided enable advertising is not in progress nor
716  * advertising is already enabled.
717  *
718  * Filter policy rules:
719  * a) Filter policy is not applicable for directed adverts.
720  * b) Only connection specific filter policy is set via ATT and based on
721  * above connection flags.
722  * - If "undirected adverts" then process connection requests from all
723  * devices.
724  * - If "Use whitelist for undirected adverts" then process connection
725  requests from devices in whitelist.
726  * c) ATT i/f is insensitive to scan specific filter policy and hence scan
727  * specific filter policy set via DM i/f
728  * (i.e. DM_HCI_ULP_SET_ADVERTISING_PARAMETERS_REQ) is considered.
729  * -----------------------------------------------------------------------------
730  *----------------------------------------------------------------------------*/

732 {
733  att_prim_t type;
734  phandle_t phandle;
735  TYPED_BD_ADDR_T addrt;
