16 /* Note that when WB-Speech was added a capability bit was added at:
17 
18  - Bit 5 in the SupportedFeatures of the SDP record
19  - Bit 7 in the BRSF AT command
20 
21  Up until this point these bitmaps were identicle.
22 
23  hfp->hf_supported_features holds the bitmap as used for the BRSF message. This macro
24  discards bits 6 and 7 and shifts the remaining bits down in their place
25  to match the new format of the SupportedFeatures SDP record bitmap.
26 */
27 #define BRSF_BITMAP_TO_SDP_BITMAP(a) (((a >> 2) & ~0x1f) | (a & 0x1f))
