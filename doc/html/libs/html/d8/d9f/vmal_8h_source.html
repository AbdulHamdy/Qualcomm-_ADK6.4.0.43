32 /*
33  * The SIZEOF_OPERATOR_MESSAGE macro is used to determine the length of a message passed to the OperatorMessage API.
34  * On legacy (Gordon, Rick) systems sizeof() always returns the number of uint16 in a structure. On Crescendo,
35  * it returns the number of bytes. This is nort what is required for crescendo.
36  * The OperatorMessage API expects the message length to always be reported in number of uint16s, regardless of platform.
37  * When sending messages that contain no paramters always declare message id to be uint16, otherwise it is likely to
38  * default to a uint32.
39  */
40 #define SIZEOF_OPERATOR_MESSAGE(msg) (uint16)(sizeof(msg)/sizeof(uint16))
41 
42 /*
43  * The SIZEOF_OPERATOR_MESSAGE_ARRAY macro is used to determine the length of
44  * a message passed to the OperatorMessage API.
45  * This differs from SIZEOF_OPERATOR_MESSAGE in that it takes the allocated
46  * size of an array of uint16 values rather than an object.
47  */
48 #define SIZEOF_OPERATOR_MESSAGE_ARRAY(s) (uint16)((s)/sizeof(uint16))
