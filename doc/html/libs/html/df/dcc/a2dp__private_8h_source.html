57 /*
58  The watchdog timeouts.
59 */
60 /*
61  The GAVDP specifies TGAVDP100 which is a
62  short response timeout for certain signals
63  in order to prevent the user seeing an
64  obvious hang.
65  The range proposed by the spec is 0.5 to
66  3 seconds.
67 
68  In practice, when the AV device is in a
69  scatternet with a SCO connection, baseband
70  interactions can stall signals excessively.
71 */
72 #ifndef A2DP_DEBUG_LIB
73 #define WATCHDOG_TGAVDP100 D_SEC(5)
74 #else
75 /* the timeout should be longer in debug mode
76  as printf takes a long time. */
77 #define WATCHDOG_TGAVDP100 D_SEC(10)
78 #endif
79 
80 /*
81  For signals not specified to use TGAVDP100,
82  we still need a timeout to prevent our device
83  hanging. This needs to be long as the spec
84  does not specify a timeout and so the remote
85  device could potentially prompt the user
86  before responding to a signal.
87 */
